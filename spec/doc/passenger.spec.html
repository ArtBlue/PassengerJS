Passenger Instantiation

1. Is virtual DOM available for manipulation
jsdom.env({
	html: oConfig.hInput
	, done: function(errors, window) {
		expect(window.document).to.be.an('object');
		done();
	}
});
2. We can instantiate a new Passenger JS instance
jsdom.env({
	html: oConfig.hInput
	, done: function(errors, window) {
	 	var password1 = new Passenger({ field: 'password' });
		expect(password1).to.exist;
		expect(password1).to.be.an('object');
		done();
    }
});
